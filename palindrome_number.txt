class Solution(object):
    def isPalindrome(self, x):
        if x < 0:
            return False
        n=x 
        s=0
        while n > 0:
            a = n % 10
            s= s * 10 + a
            n = n // 10
        
        if s==x:
            return True
        
        else:
            return False
        
       
        """
        :type x: int
        :rtype: bool
        """
